```typescript
export interface DependencyResponse {
  id: number;
  task_id: number;
  blocks_task_id: number;
  created_at: string;
}

export interface DependencyListResponse {
  blocks: DependencyResponse[];        // Tasks this task blocks
  blocked_by: DependencyResponse[];    // Tasks that block this task
}

export interface CreateDependencyInput {
  blocks_task_id: number;
}
```

**client.ts additions:**
```typescript
// Dependency management functions
export async function addDependency(
  taskId: number,
  data: CreateDependencyInput
): Promise<DependencyResponse> {
  return apiRequest<DependencyResponse>(`/api/v1/tasks/${taskId}/dependencies`, {
    method: 'POST',
    body: JSON.stringify(data),
  });
}

export async function removeDependency(
  taskId: number,
  blocksTaskId: number
): Promise<void> {
  await apiRequest<void>(`/api/v1/tasks/${taskId}/dependencies/${blocksTaskId}`, {
    method: 'DELETE',
  });
}

export async function getDependencies(taskId: number): Promise<DependencyListResponse> {
  return apiRequest<DependencyListResponse>(`/api/v1/tasks/${taskId}/dependencies`);
}
```

Follow existing API client patterns.

- Command: `dep-remove <id> <blocks-id>`
- Description: "Remove dependency (task <id> no longer blocks task <blocks-id>)"

- Call removeDependency(id, blocksId) on confirmation
- Output: JSON envelope or success message

- Argument: `<id>` (number)
