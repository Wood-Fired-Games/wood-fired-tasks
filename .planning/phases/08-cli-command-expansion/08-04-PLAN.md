```typescript
  content: string;
```

**client.ts additions:**
```typescript
// Comment management functions
export async function addComment(
  taskId: number,
  data: CreateCommentInput
): Promise<CommentResponse> {
  return apiRequest<CommentResponse>(`/api/v1/tasks/${taskId}/comments`, {
    method: 'POST',
    body: JSON.stringify(data),
  });
}

export async function getComments(taskId: number): Promise<CommentResponse[]> {
  return apiRequest<CommentResponse[]>(`/api/v1/tasks/${taskId}/comments`);
}

export async function deleteComment(commentId: number): Promise<void> {
  await apiRequest<void>(`/api/v1/comments/${commentId}`, {
    method: 'DELETE',
  });
}
```

Note: Comment delete endpoint is `/api/v1/comments/:id` (not under tasks).

**formatCommentList(comments: CommentResponse[]): string**

Format (chronological order):

- Format timestamp with new Date(created_at).toLocaleString()
- Indent comment content (2 spaces) for readability
- Separate comments with blank line
- Respect shouldUseColor(): bold timestamp and author when colors enabled
- If no comments, return "No comments"

**comment-add.ts:**
- Command: `comment-add <id>`
- Description: "Add a comment to a task"
- Argument: `<id>` - task ID (number)
- Options:
  - `-a, --author <author>` (required) - comment author name
  - `-c, --content <content>` (required) - comment text
- Interactive: prompt for missing author/content
- Behavior:
  - Call addComment(id, { author, content })
  - Output: JSON envelope or success message "Comment added to task {id}"
- Use promptForMissing() for author and content fields

**comment-list.ts:**
- Command: `comment-list <id>`
- Description: "List comments for a task"
- Argument: `<id>` - task ID (number)
- Behavior:
  - Call getComments(id)
  - Output: JSON envelope or formatCommentList()
- Show "No comments" if empty array

**comment-delete.ts:**
- Command: `comment-delete <comment-id>`
- Description: "Delete a comment by ID"
- Argument: `<comment-id>` - comment ID (number, NOT task ID)
- Behavior:
  - Call confirmAction() with comment ID (unless --force)
  - Call deleteComment(commentId) on confirmation
  - Output: JSON envelope or success message "Comment {commentId} deleted"
- Note: Can't show comment content before deletion since API doesn't have GET /comments/:id endpoint. Confirm by ID only.
